create table tbFilme(
    idFilme int PRIMARY key AUTO_INCREMENT,
    nomeFilme varchar(40),
    anoFilme int,
    generoFilme varchar(100),
    diretorFilme varchar(100),
    reviewFilme longtext

);

create table tbGenero(
    idGenero int AUTO_INCREMENT PRIMARY key,
    nome varchar(50),
    id_filme int
);

ALTER TABLE tbGenero ADD FOREIGN KEY id_filme REFERENCES tbFilme(idFilme);
